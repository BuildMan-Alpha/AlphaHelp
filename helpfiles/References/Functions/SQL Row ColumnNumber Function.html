SQL::Row::ColumnNumber()

Syntax

Index as N = .ColumnNumber( Name as C )

Argument

Description

Index

The index of the column.

A SQL::Row object created with a DIM statement.

Name

The name of a column.

Description

The ColumnNumber() method returns the index of a column.

Example

dim conn as SQL::Connection

dim rs as SQL::ResultSet

dim row as SQL::Row

dim connString as C

dim select_exp as C

connString = "{A5API='Access', FileName='c:\program files\a5v8\mdbfiles\alphasports.mdb'}"

select_exp = "select * from customer where bill_state_region = 'ny'"

if .not. conn.open(connString)

ui_msg_box("Error", conn.CallResult.text)

end

end if

if .not. conn.execute(select_exp)

ui_msg_box("Error", conn.CallResult.text)

conn.close()

end

end if

rs = conn.ResultSet

rs.NextRow()

row = rs.CurrentRow

ui_msg_box("Column Number", "" + row.ColumnNumber("lastname"))

conn.close()
