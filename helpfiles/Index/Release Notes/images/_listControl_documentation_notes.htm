How to Create a SQL Server Trigger to Set the Date/Time Field in a Table When Data is Changed

 

 

SET ANSI_NULLS ON

GO

SET QUOTED_IDENTIFIER ON

GO

 

CREATE TRIGGER [dbo].[Customer_UpdateTrigger]

ON [dbo].[Customer]

AFTER UPDATE

AS

IF NOT UPDATE(dateTimeLastUpdate)

BEGIN

-- SET NOCOUNT ON

-- interfering with SELECT statements.

SET NOCOUNT ON;

UPDATE [dbo].[Customer]

SET [dateTimeLastUpdate] = getdate()

FROM [dbo].[Customer]

JOIN inserted

ON [dbo].[Customer].CUSTOMER_ID = inserted.CUSTOMER_ID

END

set ANSI_NULLS ON

set QUOTED_IDENTIFIER ON
