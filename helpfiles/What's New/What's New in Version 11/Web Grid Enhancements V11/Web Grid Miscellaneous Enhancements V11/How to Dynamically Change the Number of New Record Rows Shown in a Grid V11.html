var rows = {grid.object}.__newRecordRows;
    if(typeof rows != 'undefined') {
$svs('NEWRECORDROWS',rows);
}

OnGridExecute

dim NewRecordRows as n
if eval_valid("Request.Variables.__si2.NewRecordRows") then 
    e.tmpl.edit.number_of_insert_rows = Request.Variables.__si2.NewRecordRows
end if

Javascript Functions

function setNewRecordRows() {
    var rows = $gvs('NEWRECORDROWS');
    var obj = {};
    obj.newRecordRows = rows;
    {Grid.Object}.setStateInfo(obj);
    {grid.object}.__newRecordRows = rows;
    {grid.object}.refresh();
}
